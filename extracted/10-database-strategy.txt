VigilAgent — Database Strategy
Version: 3.0 | Date: June 2026 | Status: Final
1. Database Strategy Overview
VigilAgent's database strategy is built on PostgreSQL as the primary data store with pgvector for
vector operations, Redis for caching and real-time data, and NATS for message queuing. The
strategy prioritizes simplicity, reliability, and cost efficiency while maintaining the ability to scale to
millions of users.
1.1 Design Principles
1. Single Source of Truth — PostgreSQL owns all persistent data
2. Cache-Aside Pattern — Redis caches frequently accessed data
3. Event-Driven Updates — Use NATS for cross-service communication
4. Read Replicas for Scaling — Separate read and write paths
5. Partitioning for Volume — Time-based partitioning for high-volume tables
2. Technology Stack

        return nil, fmt.Errorf("pinging database: %w", err) 
    }

func NewReplicaPool(primary *pgxpool.Pool, replicas []*pgxpool.Pool) *ReplicaPool { 
    return &ReplicaPool{ 
        primary:  primary, 
        replicas: replicas, 
    }
} 
// Read returns a read replica (round-robin) 
func (rp *ReplicaPool) Read() *pgxpool.Pool { 
    if len(rp.replicas) == 0 { 
        return rp.primary 
    }
     
    idx := atomic.AddUint64(&rp.counter, 1) % uint64(len(rp.replicas)) 
    return rp.replicas[idx] 
} 
// Write returns the primary 
func (rp *ReplicaPool) Write() *pgxpool.Pool { 
    return rp.primary 
} 
// Usage example 
func (rp *ReplicaPool) GetUser(ctx context.Context, userID string) (*User, error) { 
    // Read from replica 
    row := rp.Read().QueryRow(ctx, "SELECT * FROM users WHERE id = $1", userID) 
     
    var user User 
    if err := row.Scan(&user.ID, &user.Email, &user.Name); err != nil { 
        return nil, err 
    }
     
    return &user, nil 
} 
func (rp *ReplicaPool) CreateUser(ctx context.Context, user *User) error { 
    // Write to primary 
    _, err := rp.Write().Exec(ctx, 
        "INSERT INTO users (id, email, name) VALUES ($1, $2, $3)", 
        user.ID, user.Email, user.Name, 
    )
    return err 
}

func (sc *SemanticCache) Get(ctx context.Context, query string) (string, bool) { 
    // Generate embedding for query 
    embedding, err := sc.embedding.Embed(ctx, query) 
    if err != nil { 
        return "", false 
    }
     
    // Search for similar entries 
    entries, err := sc.searchSimilar(ctx, embedding) 
    if err != nil || len(entries) == 0 { 
        return "", false 
    }
     
    // Return most similar entry 
    for _, entry := range entries { 
        if entry.Similarity >= sc.similarity { 
            // Update hit count 
            sc.redis.HIncrBy(ctx, fmt.Sprintf("cache:hits:%s", entry.Key), "count", 1) 
            return entry.Value, true 
        } 
    }
     
    return "", false 
}

func (ss *SessionStore) Get(ctx context.Context, sessionID string) (*Session, error) { 
    data, err := ss.redis.Get(ctx, fmt.Sprintf("session:%s", sessionID)).Bytes() 
    if err != nil { 
        if errors.Is(err, redis.Nil) { 
            return nil, ErrSessionNotFound 
        } 
        return nil, err 
    }

    if err := json.Unmarshal(data, &session); err != nil { 
        return nil, err 
    }
     
    return &session, nil 
} 
func (ss *SessionStore) Set(ctx context.Context, session *Session) error { 
    session.ExpiresAt = time.Now().Add(ss.ttl) 
     
    data, err := json.Marshal(session) 
    if err != nil { 
        return err 
    }
     
    return ss.redis.Set(ctx, fmt.Sprintf("session:%s", session.ID), data, ss.ttl).Err() 
} 
func (ss *SessionStore) Delete(ctx context.Context, sessionID string) error { 
    return ss.redis.Del(ctx, fmt.Sprintf("session:%s", sessionID)).Err() 
}

// internal/cache/ratelimit.go 
type RateLimiter struct { 
    redis *redis.Client 
} 
type RateLimitResult struct { 
    Allowed   bool 
    Limit     int 

--- PAGE 10 ---

    Remaining int 
    ResetAt   time.Time 
} 
func (rl *RateLimiter) Allow(ctx context.Context, key string, limit int, window time.Duration
    now := time.Now() 
    windowStart := now.Truncate(window) 
    windowKey := fmt.Sprintf("ratelimit:%s:%d", key, windowStart.Unix()) 
     
    pipe := rl.redis.Pipeline() 
     
    // Increment counter 
    incr := pipe.Incr(ctx, windowKey) 
    pipe.Expire(ctx, windowKey, window) 
     
    _, err := pipe.Exec(ctx) 
    if err != nil { 
        return nil, err 
    }
     
    count := incr.Val() 
    remaining := limit - int(count) 
    if remaining < 0 { 
        remaining = 0 
    }
     
    return &RateLimitResult{ 
        Allowed:   count <= int64(limit), 
        Limit:     limit, 
        Remaining: remaining, 
        ResetAt:   windowStart.Add(window), 
    }, nil
}

func (sm *ScalingManager) ScaleReads(ctx context.Context, load float64) error { 
    // If load is high, add more replicas 
    if load > 0.8 && len(sm.replicas) < 5 { 
        return sm.addReplica(ctx) 

--- PAGE 11 ---

    }
     
    // If load is low, remove replicas 
    if load < 0.3 && len(sm.replicas) > 1 { 
        return sm.removeReplica(ctx) 
    }
     
    return nil 
} 
func (sm *ScalingManager) addReplica(ctx context.Context) error { 
    // Create new read replica 
    replica, err := sm.createReplica(ctx) 
    if err != nil { 
        return err 
    }
     
    sm.replicas = append(sm.replicas, replica) 
    sm.loadBalancer.AddBackend(replica) 
     
    return nil 
} 
func (sm *ScalingManager) removeReplica(ctx context.Context) error { 
    // Remove least-used replica 
    if len(sm.replicas) == 0 { 
        return nil 
    }
     
    replica := sm.replicas[len(sm.replicas)-1] 
    sm.replicas = sm.replicas[:len(sm.replicas)-1] 
    sm.loadBalancer.RemoveBackend(replica) 
     
    // Terminate replica 
    return replica.Close(context.Background()) 
}

func (bm *BackupManager) ScheduleBackups(cfg BackupConfig) { 
    bm.schedule = cron.New() 
     
    // Daily incremental backup 
    bm.schedule.AddFunc("0 2 * * *", func() { 
        bm.incrementalBackup(context.Background()) 
    }) 
     
    // Weekly full backup 
    bm.schedule.AddFunc("0 1 * * 0", func() { 
        bm.fullBackup(context.Background()) 
    }) 
     
    // Monthly cleanup 
    bm.schedule.AddFunc("0 3 1 * *", func() { 
        bm.cleanupOldBackups(context.Background(), cfg.RetentionDays) 
    }) 
     
    bm.schedule.Start() 
} 
func (bm *BackupManager) fullBackup(ctx context.Context) error { 
    // Create backup using pg_dump 
    cmd := exec.CommandContext(ctx, 
        "pg_dump", 
        "--host", bm.db.Config().ConnConfig.Host, 
        "--port", fmt.Sprintf("%d", bm.db.Config().ConnConfig.Port), 
        "--username", bm.db.Config().ConnConfig.User, 
        "--dbname", bm.db.Config().ConnConfig.Database, 
        "--format", "custom", 
        "--compress", "9", 
        "--file", "-", 
    )
     
    output, err := cmd.Output() 
    if err != nil { 
        return fmt.Errorf("pg_dump failed: %w", err) 
    }
     
    // Upload to S3 
    key := fmt.Sprintf("backups/full/%s/vigilagent.dump", time.Now().Format("2006-01-02")) 
    return bm.s3.Upload(ctx, bm.config.S3Bucket, key, output)

func (rm *RecoveryManager) recoverDatabase(ctx context.Context) error { 
    // 1. Stop traffic to primary 
    // 2. Promote replica to primary 
    // 3. Update connection strings 
    // 4. Verify data integrity 
     
    log.Println("Starting database recovery...") 
     
    // Promote replica 
    replica := rm.getHealthyReplica() 
    if replica == nil { 
        return fmt.Errorf("no healthy replicas available") 
    }
     
    // Update application config 
    if err := rm.updateDatabaseConfig(ctx, replica); err != nil { 
        return fmt.Errorf("updating config: %w", err) 
    }
     
    // Verify data 
    if err := rm.verifyDataIntegrity(ctx); err != nil { 
        return fmt.Errorf("data integrity check failed: %w", err) 
    }
     
    log.Println("Database recovery completed") 
    return nil 
} 
func (rm *RecoveryManager) recoverRedis(ctx context.Context) error { 
    // 1. Check if Redis is actually down 
    // 2. Try to restart Redis 
    // 3. If still down, use fallback (database only mode) 
    // 4. Rebuild cache from database 
     
    log.Println("Starting Redis recovery...") 
     
    // Check Redis health 
    if err := rm.redis.Ping(ctx).Err(); err == nil { 
        log.Println("Redis is healthy, no recovery needed") 
        return nil 
    }

    }
     
    return nil
