VigilAgent — API Contract
Version: 3.0 | Date: June 2026 | Status: Final | Base URL: 
https://api.vigilagent.com/v1
1. API Design Principles
1. RESTful by Default — Standard HTTP methods and status codes
2. JSON Everywhere — All request/response bodies are JSON
3. Pagination — All list endpoints support cursor-based pagination
4. Idempotency — POST/PUT endpoints accept Idempotency-Key header
5. Versioning — URL-based versioning (/v1/, /v2/)
6. Rate Limiting — Per-user and per-endpoint rate limits with clear headers

Webhook Verification
func verifyWebhook(payload []byte, signature string, secret string) bool {
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write(payload)
    expected := "sha256=" + hex.EncodeToString(mac.Sum(nil))
    return hmac.Equal([]byte(signature), []byte(expected))
}

6. Pagination
6.1 Cursor-Based Pagination
GET /v1/tasks?limit=20&cursor=abc123
Response:
{
  "data": [...],
  "pagination": {
    "has_more": true,
    "next_cursor": "def456",
    "prev_cursor": "xyz789",
    "total_count": 156
  }
}
6.2 Pagination Rules
• Default limit: 20, Maximum: 100
• Cursors are opaque strings (base64-encoded composite keys)
• has_more indicates if more results exist beyond current page
• total_count is optional and may be expensive for large datasets

7.3 Reconnection Strategy
Client connects → Receives events → Connection drops
    │
    ├── Store last event ID
    ├── Wait 1s (exponential backoff)
    ├── Reconnect with Last-Event-ID header
    └── Resume from where it left off

8.1 Go SDK
package main
import (
    "context"
    "fmt"
    vigil "github.com/vigilagent/go-sdk"
)
func main() {
    client := vigil.NewClient("vga_live_abc123...")
    
    ctx := context.Background()
    
    // Create a task
    task, err := client.Tasks.Create(ctx, &vigil.CreateTaskRequest{
        Title:       "Add error handling to UserService",
        Description: "Add comprehensive error handling...",
        ProjectID:   "project-123",
    })
    if err != nil {
        panic(err)
    }
    
    fmt.Printf("Task created: %s (estimated cost: $%.2f)\n", 
        task.ID, task.EstimatedCost)
    
    // Stream task execution
    events := client.Tasks.Stream(ctx, task.ID)
    for event := range events {
        switch event.Type {
        case "token_stream":
            fmt.Print(event.Data.Content)

8.2 Python SDK
from vigilagent import VigilClient
client = VigilClient(api_key="vga_live_abc123...")
# Create and execute a task
task = client.tasks.create(
    title="Add error handling to UserService",
    description="Add comprehensive error handling...",
    project_id="project-123"
)
# Stream execution
for event in client.tasks.stream(task.id):
    if event.type == "token_stream":
        print(event.data.content, end="", flush=True)
    elif event.type == "task_completed":
        print(f"\nCompleted! Cost: ${event.data.cost:.2f}")
    elif event.type == "task_failed":
        print(f"\nFailed: {event.data.error}")
