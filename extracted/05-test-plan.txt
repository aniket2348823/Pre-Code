1. Testing Philosophy
VigilAgent follows a test-first development approach with the goal of 80%+ code coverage 
across all packages. Every significant feature is developed with corresponding tests written before 
or alongside the implementation.

func TestLLMProviderContract(t *testing.T) {
    providers := []struct {
        name     string
        provider LLMProvider
    }{
        {"openai", NewOpenAIProvider(apiKey)},
        {"anthropic", NewAnthropicProvider(apiKey)},
        {"google", NewGoogleProvider(apiKey)},
    }
    for _, p := range providers {
        t.Run(p.name, func(t *testing.T) {
            ctx := context.Background()
            
            // Test Chat
            t.Run("chat returns valid response", func(t *testing.T) {
                resp, err := p.provider.Chat(ctx, &LLMRequest{
                    Messages: []Message{
                        {Role: "user", Content: "Hello"},
                    },
                })
                require.NoError(t, err)
                assert.NotEmpty(t, resp.Content)
                assert.Greater(t, resp.Tokens, 0)
                assert.Greater(t, resp.Cost, float64(0))
            })
            
            // Test Stream
            t.Run("stream returns chunks", func(t *testing.T) {
                chunks, err := p.provider.Stream(ctx, &LLMRequest{
                    Messages: []Message{
                        {Role: "user", Content: "Hello"},
                    },
                })
                require.NoError(t, err)
                
                count := 0
                for chunk := range chunks {
                    assert.NotEmpty(t, chunk.Content)
                    count++
                }
                assert.Greater(t, count, 0)
            })
            
            // Test Health Check
            t.Run("health check succeeds", func(t *testing.T) {
                err := p.provider.HealthCheck(ctx)
                assert.NoError(t, err)
            })
        })
    }
}

10.2 Test Database Reset
func resetTestDatabase(t *testing.T, db *sql.DB) {
    t.Helper()
    
    // Disable foreign key checks
    _, err := db.Exec("SET FOREIGN_KEY_CHECKS = 0")
    require.NoError(t, err)
    
    // Truncate all tables
    tables := []string{
        "audit_log",
        "usage_records",
        "tasks",
        "sessions",
        "messages",
        "skills",
        "users",
    }
    
    for _, table := range tables {
        _, err := db.Exec(fmt.Sprintf("TRUNCATE %s CASCADE", table))
        require.NoError(t, err)
    }
    
    // Re-enable foreign key checks
    _, err = db.Exec("SET FOREIGN_KEY_CHECKS = 1")
    require.NoError(t, err)
}
