operational expenditure compared to naive API usage. This is accomplished through five core 
levers: Prompt Caching, Model Routing, Batch Processing, Token Compression, and Semantic 
Caching.

2. Prompt Caching
2.1 Strategy
Prompt caching reuses Key-Value (KV) tensors from previous inference runs to avoid redundant re-
computation of static prompt prefixes. This includes system instructions, RAG context, and few-
shot examples.

type PriceTable struct {
    Prices map[string]ModelPrice

type ModelPrice struct {
    InputPer1K   float64  // Cost per 1K input tokens
    OutputPer1K  float64  // Cost per 1K output tokens
    BatchDiscount float64  // Discount for batch processing
}
func (cr *CostRouter) RouteByCost(task *Task, complexity Complexity) 
*RoutingDecision {
    // Get all available models for this complexity
    candidates := cr.getModelsForComplexity(complexity)
    
    // Estimate token usage
    inputTokens := cr.estimateInputTokens(task)
    outputTokens := cr.estimateOutputTokens(task)
    
    // Calculate cost for each candidate
    type candidate struct {
        model string
        cost  float64
    }
    
    var costs []candidate
    for _, model := range candidates {
        price := cr.priceTable.Prices[model]
        cost := (float64(inputTokens)/1000 * price.InputPer1K) +
                (float64(outputTokens)/1000 * price.OutputPer1K)
        costs = append(costs, candidate{model: model, cost: cost})
    }
    
    // Sort by cost
    sort.Slice(costs, func(i, j int) bool {
        return costs[i].cost < costs[j].cost
    })
    
    // Select cheapest that meets quality requirements
    selected := costs[0]
    
    // Check budget
    if !cr.budgetMgr.HasBudget(selected.cost) {
        // Find alternative within budget
        for _, c := range costs {
            if cr.budgetMgr.HasBudget(c.cost) {
                selected = c
                break
            }
        }
    }
    
    return &RoutingDecision{
        Model:   selected.model,
        EstCost: selected.cost,
        Reason:  "cost_optimized",
    }
}

func (bp *BatchProcessor) SubmitBatch(ctx context.Context, requests 
[]BatchRequest) (string, error) {
    // Group by model
    groups := groupByModel(requests)
    
    var batchIDs []string
    for model, batch := range groups {

4.2 Batch Eligibility Rules
func (bp *BatchProcessor) IsBatchEligible(task *Task) bool {
    // Batch eligible tasks:
    // - Non-time-sensitive (background processing)
    // - Large context window (cost savings more significant)
    // - Multiple similar tasks (can be grouped)
    
    nonUrgent := task.Priority != PriorityCritical
    hasLargeContext := task.EstimatedTokens > 5000
    isBackground := task.Type == "analysis" || task.Type == "documentation"
    
    return nonUrgent && (hasLargeContext || isBackground)
}
func (bp *BatchProcessor) EstimateBatchSavings(tasks []*Task) float64 {
    totalSavings := 0.0
    
    for _, task := range tasks {
        if bp.IsBatchEligible(task) {
            individualCost := bp.estimateCost(task)
            batchCost := individualCost * 0.5 // 50% discount
            totalSavings += individualCost - batchCost
        }
    }
    
    return totalSavings
}

func (pc *PromptCompressor) removeRedundancy(prompt string) string {
    // Remove repeated instructions
    lines := strings.Split(prompt, "\n")
    seen := make(map[string]bool)
    var unique []string
    
    for _, line := range lines {
        trimmed := strings.TrimSpace(line)
        if trimmed == "" || !seen[trimmed] {
            seen[trimmed] = true
            unique = append(unique, line)
        }
    }
    
    return strings.Join(unique, "\n")
}
func (pc *PromptCompressor) consolidateExamples(prompt string) string {
    // Find all example blocks and consolidate
    exampleRegex := regexp.MustCompile(`(?s)Examples?:\s*\n(.*?)(?=\n\n|\Z)`)
    
    return exampleRegex.ReplaceAllStringFunc(prompt, func(match string) string {
        // Keep only the most relevant examples
        examples := extractExamples(match)
        if len(examples) > 3 {
            return "Examples: " + strings.Join(examples[:3], "\n")
        }
        return match
    })
}

    if totalTokens <= co.maxTokens {
        return messages
    }
    
    // Strategy 1: Remove old messages
    optimized := co.removeOldMessages(messages, totalTokens-co.maxTokens)
    
    // Strategy 2: Compress long messages
    if co.countTokens(optimized) > co.maxTokens {
        optimized = co.compressLongMessages(optimized)
    }
    
    // Strategy 3: Summarize conversation history
    if co.countTokens(optimized) > co.maxTokens {
        optimized = co.summarizeHistory(optimized)
    }
    
    return optimized
}
func (co *ContextOptimizer) removeOldMessages(messages []Message, tokensToRemove 
int) []Message {
    if len(messages) <= 2 {
        return messages
    }
    
    // Keep system message and last N messages
    systemMsg := messages[0]
    remaining := messages[1:]
    
    removed := 0
    var kept []Message
    
    // Keep from the end, removing oldest first
    for i := len(remaining) - 1; i >= 0; i-- {
        msgTokens := estimateTokens(remaining[i].Content)
        if removed+msgTokens > tokensToRemove {
            // Can't remove more without going over
            kept = append([]Message{remaining[i]}, kept...)
        } else {
            removed += msgTokens
        }
    }
    
    return append([]Message{systemMsg}, kept...)
}

    CreatedAt   time.Time
    AccessCount int
}
func (sc *SemanticCache) Search(ctx context.Context, query string) 
(*CachedResponse, bool) {
    // Generate embedding for query
    queryEmbedding, err := sc.embedding.Embed(ctx, query)
    if err != nil {
        return nil, false
    }
    
    // Search for similar cached responses
    results, err := sc.redis.VectorSearch(ctx, "semantic:cache", queryEmbedding, 
5, sc.similarity)
    if err != nil || len(results) == 0 {
        return nil, false
    }
    
    // Return most similar result
    var cached CachedResponse
    json.Unmarshal([]byte(results[0].Payload), &cached)
    
    // Update access count
    cached.AccessCount++
    sc.updateAccessCount(ctx, cached.Query)
    
    return &cached, true
}
func (sc *SemanticCache) Store(ctx context.Context, query string, response 
*LLMResponse) error {
    // Generate embedding
    embedding, err := sc.embedding.Embed(ctx, query)
    if err != nil {
        return err
    }
    
    cached := CachedResponse{
        Query:     query,
        Response:  response.Content,
        Model:     response.Model,
        Tokens:    response.Tokens,
        Cost:      response.Cost,
        Embedding: embedding,
        CreatedAt: time.Now(),
    }
    
    // Store in Redis with vector index
    data, _ := json.Marshal(cached)
    key := fmt.Sprintf("semantic:cache:%s", generateKey(query))
    
    sc.redis.Set(ctx, key, data, 7*24*time.Hour) // 7 day TTL
    
    // Add to vector index
    sc.redis.VectorAdd(ctx, "semantic:cache", embedding, key)
    
    return nil
}

6.2 Cache Invalidation Strategy
type CacheInvalidator struct {
    cache     *SemanticCache
    ttl       time.Duration
    maxAccess int
}
func (ci *CacheInvalidator) InvalidateStale(ctx context.Context) error {
    // Remove entries older than TTL
    query := `
        SELECT key FROM semantic_cache 
        WHERE created_at < NOW() - $1::interval
        AND access_count < $2
    `
    
    keys, err := ci.cache.db.QueryContext(ctx, query, ci.ttl, ci.maxAccess)
    if err != nil {
        return err
    }
    defer keys.Close()
    
    for keys.Next() {
        var key string
        if err := keys.Scan(&key); err != nil {
            continue
        }
        ci.cache.redis.Del(ctx, key)
    }
    
    return nil
}
func (ci *CacheInvalidator) InvalidateByModel(ctx context.Context, model string) 
error {
    // When model is updated, invalidate all cached responses for that model
    query := `
        DELETE FROM semantic_cache 
        WHERE model = $1
    `
    
    _, err := ci.cache.db.ExecContext(ctx, query, model)
    return err
}

    Period       string  // "daily", "weekly", "monthly", "total"
    Limit        float64
    CurrentUsage float64
    AlertThreshold float64
    HardLimit    bool
}
func (bm *BudgetManager) CheckBudget(ctx context.Context, userID string, 
estimatedCost float64) (bool, string) {
    budgets, err := bm.getUserBudgets(ctx, userID)
    if err != nil {
        return true, "" // Allow if can't check
    }
    
    for _, budget := range budgets {
        // Check if within budget
        if budget.CurrentUsage+estimatedCost > budget.Limit {
            if budget.HardLimit {
                return false, fmt.Sprintf("Budget exceeded: $%.2f / $%.2f", 
                    budget.CurrentUsage, budget.Limit)
            }
            
            // Soft limit - alert but allow
            if budget.CurrentUsage+estimatedCost > 
budget.Limit*budget.AlertThreshold {
                bm.alerter.SendAlert(ctx, &Alert{
                    UserID:  userID,
                    Type:    "budget_warning",
                    Message: fmt.Sprintf("Budget at %.1f%%: $%.2f / $%.2f",
                        (budget.CurrentUsage/budget.Limit)*100,
                        budget.CurrentUsage,
                        budget.Limit),
                })
            }
        }
    }
    
    return true, ""
}
func (bm *BudgetManager) UpdateUsage(ctx context.Context, userID string, cost 
float64) error {
    query := `
        UPDATE budgets 
        SET current_usage = current_usage + $1,
            updated_at = NOW()
        WHERE user_id = $2 
        AND (period = 'monthly' OR period = 'daily')
    `
    
    _, err := bm.db.ExecContext(ctx, query, cost, userID)
    return err
}

func generateRecommendations(report *OptimizationReport) []string {

    if report.RoutingSavings < 200 {

classification thresholds.")
