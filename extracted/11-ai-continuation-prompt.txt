VigilAgent — AI Continuation Prompt
Version: 3.0 | Date: June 2026 | Status: Final
1. Purpose
The AI Continuation Prompt is the mechanism that enables VigilAgent agents to maintain context across
sessions, resume interrupted tasks, and continue work autonomously. It ensures that agents can pick up
exactly where they left off, even after system restarts, network failures, or user disconnections.
1.1 Core Principles
1. Zero Context Loss — Every session's state is persistently stored
2. Seamless Resumption — Agents resume without user intervention
3. Cost Efficiency — Only essential context is preserved
4. Privacy by Design — User can control what is remembered
5. Graceful Degradation — System works even with partial context

type CheckpointManager struct {

}

func (cm *CheckpointManager) CreateCheckpoint(ctx context.Context, state *SessionState) error
    // Step 1: Serialize state 
    data, err := json.Marshal(state) 
    if err != nil { 
        return fmt.Errorf("marshaling state: %w", err) 
    }
     
    // Step 2: Store in Redis (hot storage) 
    key := fmt.Sprintf("session:%s:state", state.SessionID) 
    if err := cm.redis.Set(ctx, key, data, 24*time.Hour).Err(); err != nil { 
        return fmt.Errorf("storing in redis: %w", err) 
    }
     
    // Step 3: Store in PostgreSQL (warm storage) 
    if err := cm.storeInDB(ctx, state); err != nil { 
        return fmt.Errorf("storing in database: %w", err) 
    }
     
    // Step 4: Archive to S3 (cold storage) for long-term 
    if err := cm.archiveToS3(ctx, state); err != nil { 
        // Log but don't fail - Redis and DB are sufficient 
        log.Printf("warning: failed to archive to S3: %v", err) 
    }
     
    return nil 
} 
func (cm *CheckpointManager) storeInDB(ctx context.Context, state *SessionState) error { 
    query := ` 
        INSERT INTO session_checkpoints ( 
            session_id, user_id, project_id, task_id, 
            state_data, token_count, total_cost, 
            created_at, expires_at 
        ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9) 
        ON CONFLICT (session_id)  
        DO UPDATE SET 
            state_data = EXCLUDED.state_data, 
            token_count = EXCLUDED.token_count, 
            total_cost = EXCLUDED.total_cost, 
            updated_at = NOW() 
    `
     
    data, _ := json.Marshal(state) 
     
    _, err := cm.db.Exec(ctx, query, 
        state.SessionID, state.UserID, state.ProjectID, state.TaskID, 
        data, state.TokenCount, state.TotalCost, 
        time.Now(), state.ExpiresAt, 
    )
     
    return err 
}

func (cm *CheckpointManager) archiveToS3(ctx context.Context, state *SessionState) error { 
    data, err := json.Marshal(state) 
    if err != nil { 
        return err 
    }
     
    key := fmt.Sprintf("checkpoints/%s/%s/%s.json", 
        state.UserID[:8], state.ProjectID[:8], state.SessionID) 
     
    return cm.s3.Upload(ctx, "vigilagent-checkpoints", key, data) 
}

func (rm *RecoveryManager) RecoverSession(ctx context.Context, sessionID string) (*SessionSta
    // Step 1: Try Redis (fastest) 
    state, err := rm.checkpoint.GetFromRedis(ctx, sessionID) 
    if err == nil && state != nil { 
        log.Printf("recovered session %s from Redis", sessionID) 
        return state, nil 
    }
     
    // Step 2: Try PostgreSQL 
    state, err = rm.checkpoint.GetFromDB(ctx, sessionID) 
    if err == nil && state != nil { 
        log.Printf("recovered session %s from database", sessionID) 
        // Refresh Redis cache 
        rm.checkpoint.RefreshRedis(ctx, state) 
        return state, nil 
    }
     
    // Step 3: Try S3 archive 
    state, err = rm.checkpoint.GetFromS3(ctx, sessionID) 
    if err == nil && state != nil { 
        log.Printf("recovered session %s from S3 archive", sessionID) 
        // Restore to DB and Redis 
        rm.checkpoint.StoreAll(ctx, state) 
        return state, nil 
    }

    return nil, fmt.Errorf("session %s not found", sessionID) 
} 
func (rm *RecoveryManager) ResumeExecution(ctx context.Context, state *SessionState) error { 
    // Step 1: Validate state 
    if err := rm.validateState(state); err != nil { 
        return fmt.Errorf("invalid state: %w", err) 
    }
     
    // Step 2: Check if task is still valid 
    task, err := rm.agent.GetTask(ctx, state.TaskID) 
    if err != nil { 
        return fmt.Errorf("task not found: %w", err) 
    }
     
    if task.Status == TaskCompleted || task.Status == TaskFailed { 
        return fmt.Errorf("task already %s", task.Status) 
    }
     
    // Step 3: Reconstruct context 
    context, err := rm.reconstructContext(ctx, state) 
    if err != nil { 
        return fmt.Errorf("reconstructing context: %w", err) 
    }
     
    // Step 4: Resume from checkpoint 
    return rm.agent.ResumeFromCheckpoint(ctx, state, context) 
}

func (rm *RecoveryManager) reconstructContext(ctx context.Context, state *SessionState) (*Age
    context := &AgentContext{ 
        SessionID: state.SessionID, 
        UserID:    state.UserID, 
        ProjectID: state.ProjectID, 
        TaskID:    state.TaskID, 
    }
     
    // Load project context 
    project, err := rm.agent.GetProject(ctx, state.ProjectID) 
    if err != nil { 
        return nil, fmt.Errorf("loading project: %w", err) 
    }
    context.Project = project 
     
    // Load memory 
    memory, err := rm.agent.Memory.Recall(ctx, state.TaskTitle, 10) 
    if err != nil { 
        // Continue without memory

    context.Memory = memory 
     
    // Reconstruct message history 
    context.Messages = rm.reconstructMessages(state)

func (rm *RecoveryManager) reconstructMessages(state *SessionState) []Message { 
    // Take last N messages to fit in context window 
    maxMessages := 20 
    if len(state.Messages) <= maxMessages { 
        return state.Messages 
    }
     
    // Keep system message + last N messages 
    systemMsg := state.Messages[0] 
    recentMsgs := state.Messages[len(state.Messages)-maxMessages:] 
     
    return append([]Message{systemMsg}, recentMsgs...) 
}

6.1 Context Pruning
// internal/continuation/pruning.go 
type ContextPruner struct { 
    maxTokens int 
    llm       LLMProvider 
} 
func (cp *ContextPruner) PruneContext(ctx context.Context, state *SessionState) *SessionState
    currentTokens := cp.countTokens(state.Messages) 
     
    if currentTokens <= cp.maxTokens { 
        return state 
    }
     
    // Strategy 1: Remove old tool calls and results 
    pruned := cp.removeOldToolCalls(state) 
    if cp.countTokens(pruned.Messages) <= cp.maxTokens { 
        return pruned 
    }
     
    // Strategy 2: Summarize old messages 
    pruned = cp.summarizeOldMessages(ctx, pruned) 
    if cp.countTokens(pruned.Messages) <= cp.maxTokens { 
        return pruned 
    }
     
    // Strategy 3: Keep only essential context 
    pruned = cp.keepEssentialContext(pruned) 
     

--- PAGE 10 ---

    return pruned 
} 
func (cp *ContextPruner) removeOldToolCalls(state *SessionState) *SessionState { 
    // Remove tool call details from older messages 
    var pruned []Message 
     
    for i, msg := range state.Messages { 
        if i < len(state.Messages)-10 && msg.Role == "tool" { 
            // Compress tool results 
            msg.Content = fmt.Sprintf("[Tool result: %s]", truncate(msg.Content, 100)) 
        } 
        pruned = append(pruned, msg) 
    }
     
    state.Messages = pruned 
    return state 
} 
func (cp *ContextPruner) summarizeOldMessages(ctx context.Context, state *SessionState) *Sess
    if len(state.Messages) <= 5 { 
        return state 
    }
     
    // Keep last 5 messages, summarize the rest 
    toSummarize := state.Messages[:len(state.Messages)-5] 
    toKeep := state.Messages[len(state.Messages)-5:] 
     
    // Use LLM to summarize 
    summary, err := cp.llm.Chat(ctx, &ChatRequest{ 
        Messages: []Message{ 
            {Role: "system", Content: "Summarize this conversation concisely, preserving key 
            {Role: "user", Content: formatMessages(toSummarize)}, 
        },
        MaxTokens: 500, 
    }) 
     
    if err != nil { 
        // Fallback: just keep recent messages 
        state.Messages = toKeep 
        return state 
    }
     
    // Prepend summary 
    state.Messages = append([]Message{ 
        {Role: "system", Content: fmt.Sprintf("Previous context:\n%s", summary.Content)}, 
    }, toKeep...) 
     
    return state 
}

// internal/continuation/cost.go 
type ContinuationCost struct { 
    CheckpointCost  float64  // Cost to save state 
    RecoveryCost    float64  // Cost to restore state 
    SummarizationCost float64 // Cost to summarize context 
    TotalCost       float64 
} 
func (cc *ContinuationCost) ShouldContinue(state *SessionState, estimatedCost float64) bool {
    // Don't continue if cost exceeds task budget 
    if state.TotalCost+estimatedCost > state.Budget { 
        return false 
    }
     
    // Don't continue if task has been running too long 
    if time.Since(state.StartedAt) > 30*time.Minute { 
        return false 
    }
     
    return true 
}

9. Best Practices
9.1 For Developers
1. Save Early, Save Often — Checkpoints should be created at natural breakpoints
2. Keep State Small — Only persist essential data; derive the rest
3. Idempotent Operations — Ensure resuming from a checkpoint doesn't cause duplicate work
4. Test Recovery — Regularly test recovery from various failure scenarios
5. Monitor Costs — Track continuation costs to ensure they're worth the savings
9.2 For Users
1. Let It Resume — Trust the system to continue your work
2. Review Checkpoints — Periodically review what's being saved
3. Set Budgets — Configure cost limits to prevent runaway continuations
4. Provide Feedback — Help improve continuation quality by rating resumed sessions
10. Future Enhancements
10.1 Planned Features
1. Cross-Session Learning — Continuations learn from past failures
2. Predictive Resumption — Predict when users will return and pre-warm sessions
3. Multi-Agent Continuation — Continue multiple related sessions together
4. Distributed Continuation — Continue across multiple machines
10.2 Research Areas
1. Optimal Checkpoint Frequency — Balance cost vs. recovery time
2. Context Compression — More aggressive summarization without losing quality
3. Federated Continuation — Continue sessions across edge devices
This AI Continuation Prompt system ensures VigilAgent agents can maintain context and resume work
seamlessly, providing a reliable and cost-effective experience for users.
