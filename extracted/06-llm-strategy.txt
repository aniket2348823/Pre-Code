1. LLM Strategy Overview
VigilAgent's LLM strategy is built on three pillars: Multi-Provider Resilience, Cost-Aware 
Routing, and Quality-Optimized Prompting. The goal is to deliver maximum output quality at 
minimum cost by intelligently selecting the right model for each task.
1.1 Strategic Principles
1. No Single Point of Failure — Never depend on one LLM provider
2. Cost-Performance Optimization — Route simple tasks to cheap models, complex tasks to 
powerful models
3. Graceful Degradation — Always have a fallback path when providers fail
4. Continuous Learning — Track provider performance and adjust routing dynamically
5. User Transparency — Always show which model was used and why

func (h *HealthMonitor) CheckHealth(ctx context.Context, provider string) {
    start := time.Now()
    
    err := h.providers[provider].Provider.HealthCheck(ctx)
    latency := time.Since(start)

        health.ErrorRate = math.Min(1.0, health.ErrorRate+0.1)

            health.Status = StatusDown
        } else if health.ConsecutiveFails >= 1 {
            health.Status = StatusUnhealthy
        }
    } else {
        health.ConsecutiveFails = 0
        health.ErrorRate = math.Max(0, health.ErrorRate-0.05)
        health.LatencyP50 = latency
        
        if health.ErrorRate < 0.1 {
            health.Status = StatusHealthy
        } else {
            health.Status = StatusDegraded
        }
    }
}

4.2 Failover Implementation
func (r *ModelRouter) ExecuteWithFailover(ctx context.Context, task *Task) 
(*LLMResponse, error) {
    decision, err := r.Route(ctx, task)
    if err != nil {
        return nil, fmt.Errorf("routing failed: %w", err)
    }
    
    // Try primary
    resp, err := r.executeProvider(ctx, decision.Provider, decision.Model, task)
    if err == nil {
        r.recordSuccess(decision.Provider, decision.Model, resp)
        return resp, nil
    }
    
    r.recordFailure(decision.Provider, decision.Model, err)
    
    // Try fallbacks
    for _, fallback := range decision.Fallbacks {
        resp, err := r.executeProvider(ctx, fallback.Provider, fallback.Model, 
task)
        if err == nil {
            r.recordSuccess(fallback.Provider, fallback.Model, resp)
            return resp, nil
        }
        r.recordFailure(fallback.Provider, fallback.Model, err)
    }
    
    // All providers failed — try cache
    cached, err := r.cache.Get(ctx, task.CacheKey())
    if err == nil && cached != nil {
        return cached.ToLLMResponse(), nil
    }
    
    // Queue for retry
    return nil, ErrAllProvidersFailed
}
4.3 Circuit Breaker Pattern
type CircuitBreaker struct {

--- PAGE 7 ---

    state       CircuitState
    failCount   int
    successCount int
    lastFailure time.Time
    threshold   int
    timeout     time.Duration
    mu          sync.RWMutex
}
type CircuitState int
const (
    CircuitClosed CircuitState = iota  // Normal operation
    CircuitOpen                        // Failing, reject requests
    CircuitHalfOpen                    // Testing if recovered
)
func (cb *CircuitBreaker) Execute(fn func() error) error {
    cb.mu.RLock()
    state := cb.state
    cb.mu.RUnlock()
    
    if state == CircuitOpen {
        if time.Since(cb.lastFailure) > cb.timeout {
            cb.mu.Lock()
            cb.state = CircuitHalfOpen
            cb.mu.Unlock()
        } else {
            return ErrCircuitOpen
        }
    }
    
    err := fn()
    
    cb.mu.Lock()
    defer cb.mu.Unlock()
    
    if err != nil {
        cb.failCount++
        cb.lastFailure = time.Now()
        
        if cb.failCount >= cb.threshold {
            cb.state = CircuitOpen
        }
        return err
    }
    
    cb.successCount++
    if cb.state == CircuitHalfOpen {
        cb.state = CircuitClosed
        cb.failCount = 0
    }
    return nil
}

        })
    }

7.1 Cost-Aware Execution
type CostOptimizer struct {
    budgetManager *BudgetManager
    priceTable    *PriceTable
}
func (co *CostOptimizer) OptimizeExecution(task *Task, decision 
*RoutingDecision) *ExecutionPlan {
    plan := &ExecutionPlan{
        Primary: decision,
        Steps:   []ExecutionStep{},
    }
    
    // Estimate total cost
    estimatedCost := co.estimateCost(task, decision)
    
    // Check if we can use a cheaper model
    if estimatedCost > task.Budget {
        cheaper := co.findCheaperAlternative(task, decision)
        if cheaper != nil {
            plan.Primary = cheaper
            estimatedCost = co.estimateCost(task, cheaper)
        }
    }
    
    // Add caching step if applicable
    if co.canCache(task) {
        plan.Steps = append(plan.Steps, ExecutionStep{
            Type: "cache_lookup",
            Description: "Check semantic cache for similar past tasks",
        })
    }
    
    // Add compression step if context is large
    if task.ContextSize > 10000 {
        plan.Steps = append(plan.Steps, ExecutionStep{
            Type: "context_compression",
            Description: "Compress context to reduce token usage",
        })
    }
    
    plan.EstimatedCost = estimatedCost
    return plan
}
func (co *CostOptimizer) estimateCost(task *Task, decision *RoutingDecision) 
float64 {
    price := co.priceTable.GetPrice(decision.Provider, decision.Model)
    
    inputTokens := estimateInputTokens(task)
    outputTokens := estimateOutputTokens(task)
    
    inputCost := float64(inputTokens) / 1_000_000 * price.InputPrice
    outputCost := float64(outputTokens) / 1_000_000 * price.OutputPrice
    
    return inputCost + outputCost
}

7.2 Batch Processing
type BatchProcessor struct {
    queue    *TaskQueue
    batchSize int
    interval time.Duration
}
func (bp *BatchProcessor) ProcessBatch(ctx context.Context) error {
    tasks, err := bp.queue.DequeueBatch(bp.batchSize)
    if err != nil {
        return fmt.Errorf("dequeue failed: %w", err)
    }
    
    if len(tasks) == 0 {
        return nil
    }
    
    // Group by model for efficient batching
    groups := groupByModel(tasks)
    
    for model, batch := range groups {
        // Use batch API for 50% discount
        err := bp.processBatchWithModel(ctx, model, batch)
        if err != nil {
            // Fall back to individual processing
            for _, task := range batch {
                bp.queue.Requeue(task)
            }
        }
    }
    
    return nil
}
func (bp *BatchProcessor) processBatchWithModel(ctx context.Context, model 
string, tasks []*Task) error {
    // Build batch request
    requests := make([]BatchRequest, len(tasks))
    for i, task := range tasks {
        requests[i] = BatchRequest{
            ID:      task.ID,
            Prompt:  task.BuildPrompt(),
            MaxTokens: 4096,
        }
    }
    
    // Submit to batch API
    results, err := bp.submitBatch(ctx, model, requests)
    if err != nil {
        return err
    }
    
    // Process results
    for _, result := range results {
        bp.processResult(result)
    }
    
    return nil
}

SELECT 
    model,
    SUM(cost) as total_cost,
    COUNT(*) as request_count,
    AVG(cost) as avg_cost_per_request
FROM llm_usage
WHERE recorded_at > NOW() - INTERVAL '24 hours'
GROUP BY model
ORDER BY total_cost DESC;
