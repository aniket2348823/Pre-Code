1. Architecture Overview
1.1 Design Philosophy
VigilAgent's architecture follows five core principles:
1. Interface-Driven — Every external dependency (LLM providers, databases, storage) is 
accessed through Go interfaces, enabling testing, switching, and future-proofing
2. Fail-Safe by Default — Agents never take destructive actions without explicit human 
approval; all failures are handled gracefully
3. Cost-Aware — Every LLM call tracks tokens and cost; agents have configurable budgets
4. Observable — Every action is traceable, every decision explainable, every token accounted 
for
5. Horizontally Scalable — Stateless API servers, stateful agents, distributed by design

2. Component Deep Dive
2.1 API Gateway
Technology: Go chi router with custom middleware stack
Middleware Pipeline:
Request → RequestID → Logger → CORS → RateLimiter → Auth → Handler
Key Design Decisions:
• chi chosen over Gin/Echo for stdlib compatibility and lightweight design
• Rate limiting uses token bucket algorithm (Redis-backed for distributed)
• Authentication supports both JWT (interactive) and API keys (programmatic)
• All middleware is composable and testable independently

    StatePending    TaskState = "pending"
    StatePlanning   TaskState = "planning"
    StateExecuting  TaskState = "executing"
    StateWaitingHITL TaskState = "waiting_hitl"
    StateReviewing  TaskState = "reviewing"

func (a *Agent) Transition(task *Task, event Event) error {
    switch task.State {
    case StatePending:
        if event == EventStart {
            task.State = StatePlanning
            return a.planTask(task)
        }
    case StatePlanning:
        if event == EventPlanReady {
            task.State = StateExecuting
            return a.executeNextStep(task)
        }
    case StateExecuting:
        if event == EventStepComplete {
            if a.requiresHITL(task.CurrentStep) {
                task.State = StateWaitingHITL
                return a.requestHITLApproval(task)
            }
            return a.executeNextStep(task)
        }
        if event == EventStepFailed {
            return a.handleStepFailure(task)
        }
    case StateWaitingHITL:
        if event == EventHITLApproved {
            task.State = StateExecuting
            return a.executeNextStep(task)
        }
        if event == EventHITLRejected {
            task.State = State.Failed
            return a.recordRejection(task)
        }
    case StateReviewing:
        if event == EventReviewPassed {
            task.State = StateCompleted
            return a.deliverResult(task)
        }
        if event == EventReviewFailed {
            task.State = StateExecuting
            return a.fixIssues(task)
        }
    }
    return fmt.Errorf("invalid transition: %s + %s", task.State, event)
}

type ToolRegistry struct {
    tools map[string]Tool

}
func (r *ToolRegistry) Register(tool Tool) {
    r.mu.Lock()
    defer r.mu.Unlock()
    r.tools[tool.Name()] = tool
}
func (r *ToolRegistry) Get(name string) (Tool, bool) {
    r.mu.RLock()
    defer r.mu.RUnlock()
    tool, ok := r.tools[name]
    return tool, ok
}
func (r *ToolRegistry) List() []Tool {
    r.mu.RLock()
    defer r.mu.RUnlock()
    tools := make([]Tool, 0, len(r.tools))
    for _, tool := range r.tools {
        tools = append(tools, tool)
    }
    return tools
}

type EpisodicMemory struct {
    ID        uuid.UUID
    SessionID string
    UserID    string
    Task      string
    Decision  string
    Outcome   string
    Timestamp time.Time
    Tokens    int
    Cost      float64
    Embedding []float32  // For semantic search
}
func (m *MemoryManager) Recall(ctx context.Context, query string, limit int) 
([]EpisodicMemory, error) {
    // 1. Check working memory (current session)
    results := m.working.Search(query, limit)
    if len(results) >= limit {
        return results, nil
    }
    
    // 2. Check semantic cache (Redis)
    cached, err := m.cache.SemanticSearch(ctx, query, limit-len(results))
    if err == nil && len(cached) > 0 {
        results = append(results, cached...)
    }
    if len(results) >= limit {
        return results, nil
    }
    
    // 3. Search pgvector (PostgreSQL)
    vectorResults, err := m.semantic.VectorSearch(ctx, query, limit-
len(results))
    if err != nil {
        return results, nil
    }
    results = append(results, vectorResults...)
    
    return results, nil
}

func (r *ModelRouter) Route(ctx context.Context, task *Task) (*RoutingDecision, 
error) {
    // 1. Classify task complexity
    complexity := r.classifyComplexity(task)
    
    // 2. Get available healthy providers
    available := r.healthMonitor.GetHealthyProviders()
    
    // 3. Filter by required capabilities
    capable := r.filterByCapabilities(available, task.RequiredCapabilities)
    
    // 4. Rank by cost-effectiveness
    ranked := r.rankByCostEffectiveness(capable, complexity)
    
    // 5. Select top provider
    selected := ranked[0]
    
    // 6. Check budget
    if !r.costTracker.HasBudget(ctx, selected.EstCost) {
        // Fall back to cheaper option or cached response
        return r.fallbackRoute(ctx, task, complexity)
    }
    
    return selected, nil
}
func (r *ModelRouter) classifyComplexity(task *Task) Complexity {
    // Canonical implementation — see LLM Strategy doc §3.2 for full scoring 
formula
    score := 0.0
    
    // Factor 1: Task type
    switch task.Type {
    case "formatting", "rename", "documentation":
        score += 0.1
    case "bug_fix", "small_feature":
        score += 0.3
    case "refactoring", "feature":
        score += 0.5
    case "architecture", "security":
        score += 0.7
    }
    
    // Factor 2: Scope (files affected)
    fileCount := len(task.FilesChanged)
    if fileCount > 10 {
        score += 0.3
    } else if fileCount > 5 {
        score += 0.2
    } else if fileCount > 1 {
        score += 0.1
    }
    
    // Factor 3: Reasoning depth
    if task.RequiresReasoning {
        score += 0.2
    }
    
    // Factor 4: Novelty (new patterns vs. known patterns)
    if task.IsNovel {
        score += 0.15
    }
    

--- PAGE 9 ---

    // Factor 5: Risk level
    if task.Tags.Contains("security") || task.Tags.Contains("production") {
        score += 0.3
    }
    
    return Complexity(math.Min(1.0, score))
    
    // Threshold mapping:
    // score < 0.3  → ComplexitySimple   (GPT-4o-mini, Claude Haiku)
    // score < 0.6  → ComplexityModerate (GPT-4o, Claude Sonnet)
    // score < 0.85 → ComplexityComplex  (Claude Opus, GPT-4.5)
    // score ≥ 0.85 → ComplexityCritical (Best model + HITL)
}

6.2 Distributed Tracing
// OpenTelemetry tracing
func (a *Agent) ExecuteTask(ctx context.Context, task *Task) error {
    ctx, span := tracer.Start(ctx, "agent.execute_task",
        trace.WithAttributes(
            attribute.String("task.id", task.ID),
            attribute.String("task.type", task.Type),
        ),
    )
    defer span.End()
    
    // Each step gets its own span
    ctx, planSpan := tracer.Start(ctx, "agent.plan_task")
    plan, err := a.planTask(ctx, task)
    planSpan.End()
    
    for i, step := range plan.Steps {
        ctx, stepSpan := tracer.Start(ctx, fmt.Sprintf("agent.execute_step.%d", 
i))
        result, err := a.executeStep(ctx, task, step)
        stepSpan.SetAttributes(
            attribute.String("step.tool", step.Tool),
            attribute.Int("step.tokens", result.Tokens),
            attribute.Float64("step.cost", result.Cost),
        )
        stepSpan.End()
    }
    
    return nil
}
