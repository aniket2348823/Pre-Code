func NewServer(
    cfg *config.Config,
    db *pgxpool.Pool,
    redis *redis.Client,
    llmRouter *llm.Router,
) *Server {
    // Initialize services
    authSvc := auth.NewService(db, cfg.Auth)
    agentSvc := agent.NewAgent(llmRouter, db, redis)
    metrics := telemetry.NewMetrics()
    logger := slog.New(slog.NewJSONHandler(os.Stdout, nil))
    
    // Create server
    s := &Server{
        router:  chi.NewRouter(),
        agent:   agentSvc,
        auth:    authSvc,
        metrics: metrics,
        logger:  logger,
    }
    
    // Setup routes
    s.setupRoutes()
    
    return s
}
func (s *Server) setupRoutes() {
    // Middleware
    s.router.Use(middleware.RequestID)
    s.router.Use(middleware.RealIP)
    s.router.Use(NewLoggerMiddleware(s.logger))
    s.router.Use(middleware.Recoverer)
    
    // Public routes
    s.router.Group(func(r chi.Router) {
        r.Post("/api/v1/auth/login", s.handleLogin)
        r.Post("/api/v1/auth/register", s.handleRegister)
        r.Get("/api/v1/health", s.handleHealth)
    })
    
    // Protected routes
    s.router.Group(func(r chi.Router) {
        r.Use(s.auth.Middleware())

            r.Post("/tasks", s.handleCreateTask)

// internal/database/postgres.go
func NewPostgresPool(ctx context.Context, cfg DatabaseConfig) (*pgxpool.Pool, 
error) {
    config, err := pgxpool.ParseConfig(cfg.URL)
    if err != nil {
        return nil, fmt.Errorf("parsing config: %w", err)
    }
    
    config.MaxConns = cfg.MaxConns
    config.MinConns = cfg.MinConns
    config.MaxConnLifetime = cfg.MaxConnLifetime
    config.MaxConnIdleTime = cfg.MaxConnIdleTime
    
    // Health check interval
    config.HealthCheckPeriod = 30 * time.Second
    
    pool, err := pgxpool.NewWithConfig(ctx, config)
    if err != nil {
        return nil, fmt.Errorf("creating pool: %w", err)
    }
    
    // Verify connection
    if err := pool.Ping(ctx); err != nil {
        return nil, fmt.Errorf("pinging database: %w", err)
    }
    
    return pool, nil
}

        }

--- PAGE 14 ---

        return err
    }
    
    return json.Unmarshal(data, dest)
}
func (c *Cache) Set(ctx context.Context, key string, value interface{}) error {
    data, err := json.Marshal(value)
    if err != nil {
        return err
    }
    
    return c.redis.Set(ctx, key, data, c.ttl).Err()
}
func (c *Cache) Delete(ctx context.Context, key string) error {
    return c.redis.Del(ctx, key).Err()
}
// Cache-aside pattern
func (c *Cache) GetOrCreate(ctx context.Context, key string, ttl time.Duration, 
create func() (interface{}, error)) (interface{}, error) {
    var result interface{}
    
    err := c.Get(ctx, key, &result)
    if err == nil {
        return result, nil
    }
    
    if !errors.Is(err, ErrCacheMiss) {
        return nil, err
    }
    
    // Create and cache
    result, err = create()
    if err != nil {
        return nil, err
    }
    
    if err := c.Set(ctx, key, result); err != nil {
        // Log but don't fail
        slog.Warn("failed to cache", "key", key, "error", err)
    }
    
    return result, nil
}

// internal/middleware/ratelimit.go
type RateLimiter struct {
    redis    *redis.Client
    limits   map[string]RateLimit
}
type RateLimit struct {
    Requests int
    Window   time.Duration
}
func NewRateLimiter(redis *redis.Client) *RateLimiter {
    return &RateLimiter{

--- PAGE 15 ---

        redis: redis,
        limits: map[string]RateLimit{
            "default": {Requests: 60, Window: time.Minute},
            "tasks":   {Requests: 30, Window: time.Minute},
        },
    }
}
func (rl *RateLimiter) Middleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        key := fmt.Sprintf("ratelimit:%s:%s", getUserID(r), r.URL.Path)
        
        limit := rl.limits["default"]
        
        allowed, err := rl.checkLimit(r.Context(), key, limit)
        if err != nil {
            http.Error(w, "Internal server error", 
http.StatusInternalServerError)
            return
        }
        
        if !allowed {
            w.Header().Set("Retry-After", fmt.Sprintf("%d", 
int(limit.Window.Seconds())))
            http.Error(w, "Rate limit exceeded", http.StatusTooManyRequests)
            return
        }
        
        next.ServeHTTP(w, r)
    })
}
func (rl *RateLimiter) checkLimit(ctx context.Context, key string, limit 
RateLimit) (bool, error) {
    pipe := rl.redis.Pipeline()
    
    incr := pipe.Incr(ctx, key)
    pipe.Expire(ctx, key, limit.Window)
    
    _, err := pipe.Exec(ctx)
    if err != nil {
        return false, err
    }
    
    count := incr.Val()
    return count <= int64(limit.Requests), nil
}
